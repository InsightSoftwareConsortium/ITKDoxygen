264{
265 if ((matrix.Rows() != this->Rows()) || (matrix.Cols() != this->Cols()))
266 {
267 return false;
268 }
269 bool equal = true;
272 for (unsigned int r = 0; r < this->Rows(); ++r)
273 {
274 for (unsigned int c = 0; c < this->Cols(); ++c)
275 {
276 if (Math::NotExactlyEquals(m_Matrix(r, c), matrix.m_Matrix(r, c)))
277 {
278 equal = false;
279 break;
280 }
281 }
282 }
283 return equal;
