203 template <typename T>
204 inline bool
205 ExposeMetaData(const MetaDataDictionary & Dictionary, const std::string key, T & outval)
206 {
207  auto keyIter = Dictionary.Find(key);
208  if (keyIter == Dictionary.End())
209  {
210  return false;
211  }
214  auto const * const TempMetaDataObject = dynamic_cast<MetaDataObject<T> const *>(keyIter->second.GetPointer());
215  if (TempMetaDataObject == nullptr)
216  {
217  return false;
218  }
219 
220  outval = TempMetaDataObject->GetMetaDataObjectValue();
221  return true;
222 }
