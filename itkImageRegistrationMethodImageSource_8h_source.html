94  /* Fill images with a 2D gaussian*/
95  using MovingImageIteratorType = itk::ImageRegionIteratorWithIndex<MovingImageType>;
96 
97  using FixedImageIteratorType = itk::ImageRegionIteratorWithIndex<FixedImageType>;
98 
99 
100  itk::Point<double, 2> center;

102  center[1] = (double)region.GetSize()[1] / 2.0;
103 
104  const double s = (double)region.GetSize()[0] / 2.0;
105 
106  itk::Point<double, 2> p;
107  itk::Vector<double, 2> d;
108 
109  /* Set the displacement */
110  itk::Vector<double, 2> displacement;
111  displacement[0] = m_Parameters[0];
112  displacement[1] = m_Parameters[1];
113 
114 
115  MovingImageIteratorType ri(m_MovingImage, region);
116  FixedImageIteratorType ti(m_FixedImage, region);
117  while (!ri.IsAtEnd())
118  {
119  p[0] = ri.GetIndex()[0];
120  p[1] = ri.GetIndex()[1];
121  d = p - center;
122  d += displacement;
123  const double x = d[0];

126  ri.Set(static_cast<typename MovingImageType::PixelType>(value));
