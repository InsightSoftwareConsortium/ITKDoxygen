30template <typename TInput, typename TOutput>
31class ITK_TEMPLATE_EXPORT IntensityLinearTransform
32{
33public:
34 using RealType = typename NumericTraits<TInput>::RealType;
35 IntensityLinearTransform()
36 {
37 m_Factor = 1.0;
38 m_Offset = 0.0;
39 m_Minimum = NumericTraits<TOutput>::NonpositiveMin();
40 m_Maximum = NumericTraits<TOutput>::max();
41#if defined(__GNUC__) && (__GNUC__ == 5) && (__GNUC_MINOR__ == 2) && defined(NDEBUG) && defined(__i386__)
42 m_EpsilonCompensation = static_cast<RealType>(std::numeric_limits<TOutput>::epsilon());
43 if (m_EpsilonCompensation == 0)
44 {
45 m_EpsilonCompensation = std::numeric_limits<RealType>::epsilon();
46 }
47#endif
48 }
49
50 ~IntensityLinearTransform() = default;
51 void
52 SetFactor(RealType a)
53 {
54 m_Factor = a;
55 }
56 void
57 SetOffset(RealType b)
58 {
59 m_Offset = b;
60 }
61 void
62 SetMinimum(TOutput min)
63 {
64 m_Minimum = min;
65 }
66 void
67 SetMaximum(TOutput max)
68 {
69 m_Maximum = max;
70 }
71
72 bool
73 operator==(const IntensityLinearTransform & other) const
74 {
75 return Math::ExactlyEquals(m_Factor, other.m_Factor) && Math::ExactlyEquals(m_Offset, other.m_Offset) &&
76 Math::ExactlyEquals(m_Maximum, other.m_Maximum) && Math::ExactlyEquals(m_Minimum, other.m_Minimum);
77 }
78
79 ITK_UNEQUAL_OPERATOR_MEMBER_FUNCTION(IntensityLinearTransform);
80
81 inline TOutput
82 operator()(const TInput & x) const
83 {
84#if defined(__GNUC__) && (__GNUC__ == 5) && (__GNUC_MINOR__ == 2) && defined(NDEBUG) && defined(__i386__)
85 RealType value = static_cast<RealType>(x) * m_Factor + m_Offset + m_EpsilonCompensation;
86 TOutput result = static_cast<TOutput>(value) - static_cast<TOutput>(m_EpsilonCompensation);
87#else
88 RealType value = static_cast<RealType>(x) * m_Factor + m_Offset;
89 auto result = static_cast<TOutput>(value);
90#endif
91 result = (result > m_Maximum) ? m_Maximum : result;
92 result = (result < m_Minimum) ? m_Minimum : result;
93 return result;
94 }
95
96private:
97 RealType m_Factor;
98 RealType m_Offset;
99 TOutput m_Maximum;
100 TOutput m_Minimum;
101#if defined(__GNUC__) && (__GNUC__ == 5) && (__GNUC_MINOR__ == 2) && defined(NDEBUG) && defined(__i386__)
102 RealType m_EpsilonCompensation;
103#endif
104};
