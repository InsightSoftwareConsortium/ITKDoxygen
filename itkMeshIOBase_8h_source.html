676  template <typename TOutput, typename TInput>
677  void
678  WriteBufferAsBinary(TInput * buffer, std::ofstream & outputFile, SizeValueType numberOfComponents)
679  {
680  if (typeid(TInput) == typeid(TOutput))
681  {
682  if (m_ByteOrder == IOByteOrderEnum::BigEndian && itk::ByteSwapper<TInput>::SystemIsLittleEndian())
683  {
684  itk::ByteSwapper<TInput>::SwapRangeFromSystemToBigEndian(buffer, numberOfComponents);
685  }
686  else if (m_ByteOrder == IOByteOrderEnum::LittleEndian && itk::ByteSwapper<TInput>::SystemIsBigEndian())
687  {
688  itk::ByteSwapper<TInput>::SwapRangeFromSystemToLittleEndian(buffer, numberOfComponents);
689  }
691 
692  outputFile.write(reinterpret_cast<char *>(buffer), numberOfComponents);
693  }
694  else
695  {
696  auto * data = new TOutput[numberOfComponents];
697  for (SizeValueType ii = 0; ii < numberOfComponents; ++ii)
698  {
699  data[ii] = static_cast<TOutput>(buffer[ii]);
700  }
701 
702  if (m_ByteOrder == IOByteOrderEnum::BigEndian && itk::ByteSwapper<TOutput>::SystemIsLittleEndian())
703  {
704  itk::ByteSwapper<TOutput>::SwapRangeFromSystemToBigEndian(data, numberOfComponents);
705  }
706  else if (m_ByteOrder == IOByteOrderEnum::LittleEndian && itk::ByteSwapper<TOutput>::SystemIsBigEndian())
707  {
708  itk::ByteSwapper<TOutput>::SwapRangeFromSystemToLittleEndian(data, numberOfComponents);
709  }
710 
711  outputFile.write(reinterpret_cast<char *>(data), numberOfComponents);
712  delete[] data;
713  }

743  template <typename TInput, typename TOutput>
744  void
745  ReadCellsBuffer(TInput * input, TOutput * output, CellGeometryEnum type)
746  {
747  if (input && output)
748  {
749  SizeValueType inputIndex = itk::NumericTraits<SizeValueType>::ZeroValue();
750  SizeValueType outputIndex = itk::NumericTraits<SizeValueType>::ZeroValue();
752 
753  for (SizeValueType ii = 0; ii < m_NumberOfCells; ++ii)
754  {
755  auto cellType = static_cast<CellGeometryEnum>(input[inputIndex++]);
756  auto nn = static_cast<unsigned int>(input[inputIndex++]);

761  {
762  output[outputIndex++] = static_cast<TOutput>(input[inputIndex++]);
763  }
764  }

767  inputIndex += nn;
