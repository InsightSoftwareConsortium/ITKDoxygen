28template <typename TInputPixel>
29class MorphologicalGradientHistogram
30{
31public:
32 MorphologicalGradientHistogram() = default;
33
34 ~MorphologicalGradientHistogram() = default;
35
36 inline void
37 AddBoundary()
38 {}
39
40 inline void
41 RemoveBoundary()
42 {}
43
44 using MapType = std::map<TInputPixel, SizeValueType>;
45
46 inline void
47 AddPixel(const TInputPixel & p)
48 {
49 m_Map[p]++;
50 }
51
52 inline void
53 RemovePixel(const TInputPixel & p)
54 {
55 m_Map[p]--;
56 }
57
58 inline TInputPixel
59 GetValue(const TInputPixel &)
60 {
61 return GetValue();
62 }
63
64 inline TInputPixel
65 GetValue()
66 {
67 // clean the map
68 typename MapType::iterator mapIt = m_Map.begin();
69 while (mapIt != m_Map.end())
70 {
71 if (mapIt->second == 0)
72 {
73 // this value must be removed from the histogram
74 // The value must be stored and the iterator updated before removing the
75 // value
76 // or the iterator is invalidated.
77 TInputPixel toErase = mapIt->first;
78 ++mapIt;
79 m_Map.erase(toErase);
80 }
81 else
82 {
83 ++mapIt;
84 }
85 }
86
87 // and return the value
88 if (!m_Map.empty())
89 {
90 return m_Map.rbegin()->first - m_Map.begin()->first;
91 }
92 return 0;
93 }
94
95 static bool
96 UseVectorBasedAlgorithm()
97 {
98 return false;
99 }
100
101 MapType m_Map;
102};

105template <typename TInputPixel>
106class VectorMorphologicalGradientHistogram
107{
108public:
109 VectorMorphologicalGradientHistogram()
110 {
111 // initialize members need for the vector based algorithm
112 m_Vector.resize(NumericTraits<TInputPixel>::max() - NumericTraits<TInputPixel>::NonpositiveMin() + 1, 0);
113 m_Max = NumericTraits<TInputPixel>::NonpositiveMin();
114 m_Min = NumericTraits<TInputPixel>::max();
115 m_Count = 0;
116 }
117
118 ~VectorMorphologicalGradientHistogram() = default;
119
120 inline void
121 AddBoundary()
122 {}
123
124 inline void
125 RemoveBoundary()
126 {}
127
128
129 inline void
130 AddPixel(const TInputPixel & p)
131 {
132 m_Vector[p - NumericTraits<TInputPixel>::NonpositiveMin()]++;
133 if (p > m_Max)
134 {
135 m_Max = p;
136 }
137 if (p < m_Min)
138 {
139 m_Min = p;
140 }
141 ++m_Count;
142 }
143
144 inline void
145 RemovePixel(const TInputPixel & p)
146 {
147 m_Vector[p - NumericTraits<TInputPixel>::NonpositiveMin()]--;
148 --m_Count;
149 if (m_Count > 0)
150 {
151 while (m_Vector[m_Max - NumericTraits<TInputPixel>::NonpositiveMin()] == 0)
152 {
153 --m_Max;
154 }
155 while (m_Vector[m_Min - NumericTraits<TInputPixel>::NonpositiveMin()] == 0)
156 {
157 ++m_Min;
158 }
159 }
160 else
161 {
162 m_Max = NumericTraits<TInputPixel>::NonpositiveMin();
163 m_Min = NumericTraits<TInputPixel>::max();
164 }
165 }
166
167 inline TInputPixel
168 GetValue(const TInputPixel &)
169 {
170 return GetValue();
171 }
172
173 inline TInputPixel
174 GetValue()
175 {
176 if (m_Count > 0)
177 {
178 return m_Max - m_Min;
179 }
180 else
181 {
182 return NumericTraits<TInputPixel>::ZeroValue();
183 }
184 }
185
186 static bool
187 UseVectorBasedAlgorithm()
188 {
189 return true;
190 }
191
192 std::vector<SizeValueType> m_Vector;
193 TInputPixel m_Min;
194 TInputPixel m_Max;
195 SizeValueType m_Count;
196};
