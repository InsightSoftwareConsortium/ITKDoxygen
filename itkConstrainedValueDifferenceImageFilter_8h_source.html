57template <typename TInput1, typename TInput2, typename TOutput>
58class ConstrainedValueDifference
59{
60public:
61 bool
62 operator==(const ConstrainedValueDifference &) const
63 {
64 return true;
65 }
66
67 ITK_UNEQUAL_OPERATOR_MEMBER_FUNCTION(ConstrainedValueDifference);
68
69 inline TOutput
70 operator()(const TInput1 & A, const TInput2 & B) const
71 {
72 const auto dA = static_cast<double>(A);
73 const auto dB = static_cast<double>(B);
74 const double diff = dA - dB;
75 const double cdiff1 =
76 diff > NumericTraits<TOutput>::NonpositiveMin() ? diff : NumericTraits<TOutput>::NonpositiveMin();
77 const double cdiff2 = cdiff1 < NumericTraits<TOutput>::max() ? cdiff1 : NumericTraits<TOutput>::max();
78
79 return static_cast<TOutput>(cdiff2);
80 }
81};
82} // namespace Functor
