181 using VisitorPointer = typename VisitorType::Pointer;
182 using VisitorPointerValueType = typename std::map<CellGeometryEnum, VisitorPointer>::value_type;

184 public:
185 VisitorType *
186 GetVisitor(CellGeometryEnum id)
187 {
188 if (id < CellGeometryEnum::LAST_ITK_CELL)
189 {
190 return m_Visitors[static_cast<int>(id)];
191 }
192 else
193 {
194 auto pos = m_UserDefined.find(id);
195 if (pos != m_UserDefined.end())
196 {
197 return pos->second;
198 }
199 }
200 return nullptr;
201 }
202
203 void
204 AddVisitor(VisitorType * v)
205 {
206 CellGeometryEnum id = v->GetCellTopologyId();
207
208 if (id < CellGeometryEnum::LAST_ITK_CELL)
209 {
210 m_Visitors[static_cast<int>(id)] = v;
211 }
212 else
213 {
214 m_UserDefined.insert(VisitorPointerValueType(id, v));
215 }
216 }
