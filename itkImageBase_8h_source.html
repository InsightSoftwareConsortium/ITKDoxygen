454 template <typename TCoordRep>
455 [[nodiscard]] IndexType
456 TransformPhysicalPointToIndex(const Point<TCoordRep, VImageDimension> & point) const
457 {
458 IndexType index;
461 for (unsigned int i = 0; i < VImageDimension; ++i)
462 {
463 TCoordRep sum{};
464 for (unsigned int j = 0; j < VImageDimension; ++j)
465 {
466 sum += this->m_PhysicalPointToIndex[i][j] * (point[j] - this->m_Origin[j]);
467 }
468 index[i] = Math::RoundHalfIntegerUp<IndexValueType>(sum);
469 }
470 return index;
471 }

507 template <typename TIndexRep, typename TCoordRep>
508 [[nodiscard]] ContinuousIndex<TIndexRep, VImageDimension>
509 TransformPhysicalPointToContinuousIndex(const Point<TCoordRep, VImageDimension> & point) const
510 {
511 ContinuousIndex<TIndexRep, VImageDimension> index;
512 Vector<SpacePrecisionType, VImageDimension> cvector;
513
514 for (unsigned int k = 0; k < VImageDimension; ++k)
515 {
516 cvector[k] = point[k] - this->m_Origin[k];
517 }
518 cvector = m_PhysicalPointToIndex * cvector;
519 for (unsigned int i = 0; i < VImageDimension; ++i)
520 {
521 index[i] = static_cast<TIndexRep>(cvector[i]);
522 }
523 return index;
524 }

550 template <typename TCoordRep, typename TIndexRep>
551 void
552 TransformContinuousIndexToPhysicalPoint(const ContinuousIndex<TIndexRep, VImageDimension> & index,
553 Point<TCoordRep, VImageDimension> & point) const
554 {
555 for (unsigned int r = 0; r < VImageDimension; ++r)
556 {
557 TCoordRep sum{};
558 for (unsigned int c = 0; c < VImageDimension; ++c)
559 {
560 sum += this->m_IndexToPhysicalPoint(r, c) * index[c];
561 }
562 point[r] = sum + this->m_Origin[r];
563 }
564 }

586 template <typename TCoordRep>
587 void
588 TransformIndexToPhysicalPoint(const IndexType & index, Point<TCoordRep, VImageDimension> & point) const
589 {
590 for (unsigned int i = 0; i < VImageDimension; ++i)
591 {
592 point[i] = this->m_Origin[i];
593 for (unsigned int j = 0; j < VImageDimension; ++j)
594 {
595 point[i] += m_IndexToPhysicalPoint[i][j] * index[j];
596 }
597 }

630 template <typename TCoordRep>
631 void
632 TransformLocalVectorToPhysicalVector(const FixedArray<TCoordRep, VImageDimension> & inputGradient,
633 FixedArray<TCoordRep, VImageDimension> & outputGradient) const
634 {
635 const DirectionType & direction = this->GetDirection();
636
637 itkAssertInDebugAndIgnoreInReleaseMacro(inputGradient.GetDataPointer() != outputGradient.GetDataPointer());
638
639 for (unsigned int i = 0; i < VImageDimension; ++i)
640 {
641 using CoordSumType = typename NumericTraits<TCoordRep>::AccumulateType;
642 CoordSumType sum{};
643 for (unsigned int j = 0; j < VImageDimension; ++j)
644 {
645 sum += direction[i][j] * inputGradient[j];
646 }
647 outputGradient[i] = static_cast<TCoordRep>(sum);
648 }
649 }
