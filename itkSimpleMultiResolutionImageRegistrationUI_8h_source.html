110  if (!optimizer)
111  return;
112 
113  unsigned int level = this->m_Registrator->GetCurrentLevel();
114  if (m_NumberOfIterations.Size() >= level + 1)
115  {
116  optimizer->SetNumberOfIterations(m_NumberOfIterations[level]);
117  }
118 
119  if (m_LearningRates.Size() >= level + 1)
120  {
121  optimizer->SetLearningRate(m_LearningRates[level]);
122  }
123 
124  std::cout << " No. Iterations: " << optimizer->GetNumberOfIterations()
125  << " Learning rate: " << optimizer->GetLearningRate() << std::endl;
126  }
