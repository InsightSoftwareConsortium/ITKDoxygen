26// The following classes are examples of simple user interface
27// that controls a MultiResolutionImageRegistrationMethod process
28
29template <typename TRegistrator>
30class SimpleMultiResolutionImageRegistrationUI
31{
32public:
33 SimpleMultiResolutionImageRegistrationUI(TRegistrator * ptr)
34 : m_Tag(0)
35 {
36
37 if (!ptr)
38 {
39 return;
40 }
41 m_Registrator = ptr;
42 typename itk::SimpleMemberCommand<SimpleMultiResolutionImageRegistrationUI>::Pointer iterationCommand =
43 itk::SimpleMemberCommand<SimpleMultiResolutionImageRegistrationUI>::New();
44
45 iterationCommand->SetCallbackFunction(this, &SimpleMultiResolutionImageRegistrationUI::StartNewLevel);
46
47 m_Tag = m_Registrator->AddObserver(itk::IterationEvent(), iterationCommand);
48 }
49
50 virtual ~SimpleMultiResolutionImageRegistrationUI()
51 {
52 if (m_Registrator)
53 {
54 m_Registrator->RemoveObserver(m_Tag);
55 }
56 }
57
58 virtual void
59 StartNewLevel()
60 {
61 std::cout << "--- Starting level " << m_Registrator->GetCurrentLevel() << std::endl;
62 }
63
64protected:
65 typename TRegistrator::Pointer m_Registrator{};
66 unsigned long m_Tag{};
67};

70// This UI supports registration methods with gradient descent
71// type optimizers.
72// This UI allows the number of iterations and learning rate
73// to be changes at each resolution level.
74template <typename TRegistration>
75class ITK_TEMPLATE_EXPORT SimpleMultiResolutionImageRegistrationUI2
76 : public SimpleMultiResolutionImageRegistrationUI<TRegistration>
77{
78public:
79 using Superclass = SimpleMultiResolutionImageRegistrationUI<
80 itk::MultiResolutionImageRegistrationMethod<itk::Image<float, 3>, itk::Image<float, 3>>>;
81
82 SimpleMultiResolutionImageRegistrationUI2(TRegistration * ptr)
83 : Superclass(ptr){};
84 ~SimpleMultiResolutionImageRegistrationUI2() override = default;
85
86 void
87 SetNumberOfIterations(itk::Array<unsigned int> & iter)
88 {
89 m_NumberOfIterations = iter;
90 }
91
92 void
93 SetLearningRates(itk::Array<double> & rates)
94 {
95 m_LearningRates = rates;
96 }
97
98 void
99 StartNewLevel() override
100 {
101
102 // call the superclass's implementation
103 this->Superclass::StartNewLevel();
104
105 if (!this->m_Registrator)
106 {
107 return;
108 }
109
110 // Try to cast the optimizer to a gradient descent type,
111 // return if casting didn't work.
112 itk::GradientDescentOptimizer::Pointer optimizer =
113 dynamic_cast<itk::GradientDescentOptimizer *>(this->m_Registrator->GetModifiableOptimizer());
114 if (!optimizer)
115 {
116 return;
117 }
118
119 unsigned int level = this->m_Registrator->GetCurrentLevel();
120 if (m_NumberOfIterations.Size() >= level + 1)
121 {
122 optimizer->SetNumberOfIterations(m_NumberOfIterations[level]);
123 }
124
125 if (m_LearningRates.Size() >= level + 1)
126 {
127 optimizer->SetLearningRate(m_LearningRates[level]);
128 }
129
130 std::cout << " No. Iterations: " << optimizer->GetNumberOfIterations()
131 << " Learning rate: " << optimizer->GetLearningRate() << std::endl;
132 }
133
134private:
135 itk::Array<unsigned int> m_NumberOfIterations{};
136 itk::Array<double> m_LearningRates{};
137};
