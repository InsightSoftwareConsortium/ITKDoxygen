144 PointType
145 ComputeOptimalLocation(const PointType & iP)
146 {
147 ComputeAMatrixAndBVector();
148
149 vnl_svd<CoordType> svd(m_A, m_SVDAbsoluteThreshold);
150 svd.zero_out_relative(m_SVDRelativeThreshold);
151
152 m_Rank = svd.rank();
153
154 const auto y = (m_B.as_vector() - m_A * iP.GetVnlVector());
155
156 VNLVectorType displacement = svd.solve(y);
157
158 PointType oP;
159 for (unsigned int dim = 0; dim < PointDimension; ++dim)
160 {
161 oP[dim] = iP[dim] + displacement[dim];
162 }
163 return oP;
164 }

182 void
183 AddPoint(const PointType & iP, const VectorType & iN, const CoordType & iWeight = static_cast<CoordType>(1.))
184 {
185 unsigned int k(0);
186 unsigned int dim1;
187 unsigned int dim2;
188
189 CoordType d = -iN * iP.GetVectorFromOrigin();
190
191 for (dim1 = 0; dim1 < PointDimension; ++dim1)
192 {
193 for (dim2 = dim1; dim2 < PointDimension; ++dim2)
194 {
195 this->m_Coefficients[k++] += iWeight * iN[dim1] * iN[dim2];
196 }
197 this->m_Coefficients[k++] += iWeight * iN[dim1] * d;
198 }
199
200 this->m_Coefficients[k++] += iWeight * d * d;
201 }
