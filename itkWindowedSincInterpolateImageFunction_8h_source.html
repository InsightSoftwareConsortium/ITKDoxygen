107template <unsigned int VRadius, typename TInput = double, typename TOutput = double>
108class ITK_TEMPLATE_EXPORT LanczosWindowFunction
109{
110public:
111 inline TOutput
112 operator()(const TInput & A) const
113 {
114 if (A == 0.0)
115 {
116 return static_cast<TOutput>(1.0);
117 }
118 double z = m_Factor * A;
119 return static_cast<TOutput>(std::sin(z) / z);
120 }
123private:
125 static const double m_Factor;
126};
