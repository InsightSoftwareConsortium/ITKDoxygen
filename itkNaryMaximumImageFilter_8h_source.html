33template <typename TInput, typename TOutput>
34class Maximum1
35{
36public:
37 using OutputValueType = typename NumericTraits<TOutput>::ValueType;
38 // not sure if this type alias really makes things more clear... could just use
39 // TOutput?
40
41 Maximum1() = default;
42 ~Maximum1() = default;
43 inline TOutput
44 operator()(const std::vector<TInput> & B) const
45 {
46 OutputValueType A = NumericTraits<TOutput>::NonpositiveMin();
47
48 for (unsigned int i = 0; i < B.size(); ++i)
49 {
50 if (A < static_cast<OutputValueType>(B[i]))
51 {
52 A = static_cast<OutputValueType>(B[i]);
53 }
54 }
55 return A;
56 }
