34template <typename TInput, typename TMask, typename TOutput = TInput>
35class MaskNegatedInput
36{
37public:
38 using AccumulatorType = typename NumericTraits<TInput>::AccumulateType;
39
40 MaskNegatedInput()
41 : m_OutsideValue(NumericTraits<TOutput>::ZeroValue())
42 , m_MaskingValue(NumericTraits<TMask>::ZeroValue())
43 {}
44 ~MaskNegatedInput() = default;
45 bool
46 operator==(const MaskNegatedInput &) const
47 {
48 return true;
49 }
50
51 ITK_UNEQUAL_OPERATOR_MEMBER_FUNCTION(MaskNegatedInput);
52
53 inline TOutput
54 operator()(const TInput & A, const TMask & B) const
55 {
56 if (B != m_MaskingValue)
57 {
58 return m_OutsideValue;
59 }
60 else
61 {
62 return static_cast<TOutput>(A);
63 }
64 }
65
67 void
68 SetOutsideValue(const TOutput & outsideValue)
69 {
70 m_OutsideValue = outsideValue;
71 }
72
74 const TOutput &
75 GetOutsideValue() const
76 {
77 return m_OutsideValue;
78 }
79
81 void
82 SetMaskingValue(const TMask & maskingValue)
83 {
84 m_MaskingValue = maskingValue;
85 }
86
88 const TMask &
89 GetMaskingValue() const
90 {
91 return m_MaskingValue;
92 }
93
94private:
95 TOutput m_OutsideValue;
96 TMask m_MaskingValue;
97};
