49template <typename TInputPixel, typename TAccumulate>
50class StandardDeviationAccumulator
51{
52public:
53 using RealType = typename NumericTraits<TInputPixel>::RealType;
54
55 StandardDeviationAccumulator(SizeValueType size)
56 {
57 m_Size = size;
58 m_Values.reserve(size);
59 }
60
61 ~StandardDeviationAccumulator() = default;
62
63 inline void
64 Initialize()
65 {
66 m_Sum = NumericTraits<TAccumulate>::ZeroValue();
67 m_Values.clear();
68 }
69
70 inline void
71 operator()(const TInputPixel & input)
72 {
73 m_Sum = m_Sum + input;
74 m_Values.push_back(input);
75 }
76
77 inline RealType
78 GetValue()
79 {
80 // to avoid division by zero
81 if (m_Size <= 1)
82 {
83 return NumericTraits<RealType>::ZeroValue();
84 }
85
86 typename NumericTraits<TInputPixel>::RealType mean = ((RealType)m_Sum) / m_Size;
87 typename std::vector<TInputPixel>::iterator it;
88 RealType squaredSum = NumericTraits<RealType>::ZeroValue();
89 for (it = m_Values.begin(); it != m_Values.end(); ++it)
90 {
91 squaredSum += itk::Math::sqr(*it - mean);
92 }
93 return std::sqrt(squaredSum / (m_Size - 1));
94 }
95
96 TAccumulate m_Sum;
97 SizeValueType m_Size;
98 std::vector<TInputPixel> m_Values;
99};
100} // namespace Functor
