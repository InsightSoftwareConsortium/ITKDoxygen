52template <typename TInput, typename TOutput>
53class Sigmoid
54{
55public:
56 Sigmoid()
57 {
58 m_Alpha = 1.0;
59 m_Beta = 0.0;
60 m_OutputMinimum = NumericTraits<TOutput>::min();
61 m_OutputMaximum = NumericTraits<TOutput>::max();
62 }
63
64 ~Sigmoid() = default;
65
66
67 bool
68 operator==(const Sigmoid & other) const
69 {
70 return Math::ExactlyEquals(m_Alpha, other.m_Alpha) && Math::ExactlyEquals(m_Beta, other.m_Beta) &&
71 Math::ExactlyEquals(m_OutputMaximum, other.m_OutputMaximum) &&
72 Math::ExactlyEquals(m_OutputMinimum, other.m_OutputMinimum);
73 }
74
75 ITK_UNEQUAL_OPERATOR_MEMBER_FUNCTION(Sigmoid);
76
77 inline TOutput
78 operator()(const TInput & A) const
79 {
80 const double x = (static_cast<double>(A) - m_Beta) / m_Alpha;
81 const double e = 1.0 / (1.0 + std::exp(-x));
82 const double v = (m_OutputMaximum - m_OutputMinimum) * e + m_OutputMinimum;
83
84 return static_cast<TOutput>(v);
85 }

88 SetAlpha(double alpha)
89 {
90 m_Alpha = alpha;
91 }
92
93 void
94 SetBeta(double beta)
95 {
96 m_Beta = beta;
97 }

100 GetAlpha() const
101 {
102 return m_Alpha;
103 }

106 GetBeta() const
107 {
108 return m_Beta;
109 }

112 SetOutputMinimum(TOutput min)

118 SetOutputMaximum(TOutput max)

120 m_OutputMaximum = max;
121 }

124 GetOutputMinimum() const

127 }

129 TOutput
130 GetOutputMaximum() const
131 {
132 return m_OutputMaximum;
133 }
134
135private:
136 double m_Alpha;
137 double m_Beta;
138 TOutput m_OutputMinimum;
139 TOutput m_OutputMaximum;
140};
