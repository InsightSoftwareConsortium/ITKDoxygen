135 template <typename T>
136 void
137 UpdateCellInformation(T * buffer)
138 {
139 unsigned int numberOfVertices = 0;
140 unsigned int numberOfVertexIndices = 0;
141 unsigned int numberOfLines = 0;
142 SizeValueType numberOfLineIndices = 0;
143 unsigned int numberOfPolygons = 0;
144 unsigned int numberOfPolygonIndices = 0;
145
146 SizeValueType index = 0;
147
148 for (SizeValueType ii = 0; ii < this->m_NumberOfCells; ++ii)
149 {
150 auto cellType = static_cast<CellGeometryEnum>(static_cast<int>(buffer[index++]));
151 auto nn = static_cast<unsigned int>(buffer[index++]);
152 switch (cellType)
153 {
154 case CellGeometryEnum::VERTEX_CELL:
155 numberOfVertices++;
156 numberOfVertexIndices += nn + 1;
157 break;
158 case CellGeometryEnum::LINE_CELL:
159 numberOfLines++;
160 numberOfLineIndices += nn + 1;
161 break;
162 case CellGeometryEnum::POLYLINE_CELL:
163 numberOfLines++;
164 numberOfLineIndices += nn + 1;
165 break;
166 case CellGeometryEnum::TRIANGLE_CELL:
167 numberOfPolygons++;
168 numberOfPolygonIndices += nn + 1;
169 break;
170 case CellGeometryEnum::POLYGON_CELL:
171 numberOfPolygons++;
172 numberOfPolygonIndices += nn + 1;
173 break;
174 case CellGeometryEnum::QUADRILATERAL_CELL:
175 numberOfPolygons++;
176 numberOfPolygonIndices += nn + 1;
177 break;
178 default:
179 itkExceptionMacro(<< "Currently we dont support this cell type");
180 }
181
182 index += nn;
183 }
184
185 MetaDataDictionary & metaDic = this->GetMetaDataDictionary();
186 EncapsulateMetaData<unsigned int>(metaDic, "numberOfVertices", numberOfVertices);
187 EncapsulateMetaData<unsigned int>(metaDic, "numberOfVertexIndices", numberOfVertexIndices);
188 EncapsulateMetaData<unsigned int>(metaDic, "numberOfLines", numberOfLines);
189 EncapsulateMetaData<unsigned int>(metaDic, "numberOfLineIndices", numberOfLineIndices);
190 EncapsulateMetaData<unsigned int>(metaDic, "numberOfPolygons", numberOfPolygons);
191 EncapsulateMetaData<unsigned int>(metaDic, "numberOfPolygonIndices", numberOfPolygonIndices);
192 return;
193 }

450 for (SizeValueType ii = 0; ii < this->m_NumberOfPoints; ++ii)
451 {
452 for (unsigned int jj = 0; jj < this->m_PointDimension - 1; ++jj)
453 {

455 }
456
457 outputFile << ConvertNumberToString(buffer[ii * this->m_PointDimension + this->m_PointDimension - 1]) << '\n';
458 }

477 template <typename T>
478 void
479 WriteCellsBufferAsASCII(std::ofstream & outputFile, T * buffer)
480 {
481 MetaDataDictionary & metaDic = this->GetMetaDataDictionary();
482 unsigned int numberOfVertices = 0;
483 unsigned int numberOfVertexIndices = 0;
484 unsigned int numberOfLines = 0;
485 unsigned int numberOfLineIndices = 0;
486 unsigned int numberOfPolygons = 0;
487 unsigned int numberOfPolygonIndices = 0;
488
490 SizeValueType index = 0;
491
492 ExposeMetaData<unsigned int>(metaDic, "numberOfVertices", numberOfVertices);
493 if (numberOfVertices)
494 {
495 ExposeMetaData<unsigned int>(metaDic, "numberOfVertexIndices", numberOfVertexIndices);
496 outputFile << "VERTICES " << numberOfVertices << " " << numberOfVertexIndices << '\n';
497 for (SizeValueType ii = 0; ii < this->m_NumberOfCells; ++ii)
498 {
499 auto cellType = static_cast<CellGeometryEnum>(static_cast<int>(buffer[index++]));
500 auto nn = static_cast<unsigned int>(buffer[index++]);
501 if (cellType == CellGeometryEnum::VERTEX_CELL)
502 {
503 outputFile << nn;
504 for (unsigned int jj = 0; jj < nn; ++jj)
505 {
506 outputFile << " " << buffer[index++];
507 }
508 outputFile << '\n';
509 }
510 else
511 {
512 index += nn;
513 }
514 }
515 }
516
518 index = 0;
519 ExposeMetaData<unsigned int>(metaDic, "numberOfLines", numberOfLines);
520 if (numberOfLines)
521 {
522 numberOfLineIndices = 0;
523 SizeValueType numberOfPolylines = 0;
524 PolylinesContainerPointer polylines = PolylinesContainerType::New();
525 PointIdVector pointIds;
526 for (SizeValueType ii = 0; ii < this->m_NumberOfCells; ++ii)
527 {
528 auto cellType = static_cast<CellGeometryEnum>(static_cast<int>(buffer[index++]));
529 auto nn = static_cast<unsigned int>(buffer[index++]);
532 pointIds.clear();
533 if (cellType == CellGeometryEnum::LINE_CELL)
534 {
535 pointIds.push_back(static_cast<SizeValueType>(buffer[index]));
536 pointIds.push_back(static_cast<SizeValueType>(buffer[index + 1]));
537 }
538 else if (cellType == CellGeometryEnum::POLYLINE_CELL)
539 {
540 for (unsigned int jj = 0; jj < nn; ++jj)
541 {
542 pointIds.push_back(static_cast<SizeValueType>(buffer[index + jj]));
543 }
544 }
545
546 polylines->InsertElement(numberOfPolylines++, pointIds);
547 numberOfLineIndices += pointIds.size();
548 index += nn;
549 }
550
551 numberOfLines = polylines->Size();
552 numberOfLineIndices += numberOfLines;
553 EncapsulateMetaData<unsigned int>(metaDic, "numberOfLines", numberOfLines);
554 EncapsulateMetaData<unsigned int>(metaDic, "numberOfLineIndices", numberOfLineIndices);
555 outputFile << "LINES " << numberOfLines << " " << numberOfLineIndices << '\n';
556 for (SizeValueType ii = 0; ii < polylines->Size(); ++ii)
557 {
558 auto nn = static_cast<unsigned int>(polylines->ElementAt(ii).size());
559 outputFile << nn;
560 for (unsigned int jj = 0; jj < nn; ++jj)
561 {
562 outputFile << " " << polylines->ElementAt(ii)[jj];
563 }
564 outputFile << '\n';
565 }
566 }
567
569 index = 0;
570 ExposeMetaData<unsigned int>(metaDic, "numberOfPolygons", numberOfPolygons);
571 if (numberOfPolygons)
572 {
573 ExposeMetaData<unsigned int>(metaDic, "numberOfPolygonIndices", numberOfPolygonIndices);
574 outputFile << "POLYGONS " << numberOfPolygons << " " << numberOfPolygonIndices << '\n';
575 for (SizeValueType ii = 0; ii < this->m_NumberOfCells; ++ii)
576 {
577 auto cellType = static_cast<CellGeometryEnum>(static_cast<int>(buffer[index++]));
578 auto nn = static_cast<unsigned int>(buffer[index++]);
579 if (cellType == CellGeometryEnum::POLYGON_CELL || cellType == CellGeometryEnum::TRIANGLE_CELL ||
580 cellType == CellGeometryEnum::QUADRILATERAL_CELL)
581 {
582 outputFile << nn;
583 for (unsigned int jj = 0; jj < nn; ++jj)
584 {
585 outputFile << " " << buffer[index++];
586 }
587 outputFile << '\n';
588 }
589 else
590 {
591 index += nn;
592 }
593 }
594 }

691 outputFile << "\n";

697 template <typename T>
698 void
699 WritePointDataBufferAsASCII(std::ofstream & outputFile, T * buffer, const StringType & pointPixelComponentName)
700 {
701 MetaDataDictionary & metaDic = this->GetMetaDataDictionary();
702 StringType dataName;
703
704 outputFile << "POINT_DATA " << this->m_NumberOfPointPixels << '\n';
705 switch (this->m_PointPixelType)
706 {
707 case IOPixelEnum::SCALAR:
708 {
709 outputFile << "SCALARS ";
710 ExposeMetaData<StringType>(metaDic, "pointScalarDataName", dataName);
711 outputFile << dataName << " ";
712 break;
713 }
714 case IOPixelEnum::OFFSET:
715 case IOPixelEnum::POINT:
716 case IOPixelEnum::COVARIANTVECTOR:
717 case IOPixelEnum::VECTOR:
718 {
719 outputFile << "VECTORS ";
720 ExposeMetaData<StringType>(metaDic, "pointVectorDataName", dataName);
721 outputFile << dataName << " ";
722 break;
723 }
724 case IOPixelEnum::SYMMETRICSECONDRANKTENSOR:
725 case IOPixelEnum::DIFFUSIONTENSOR3D:
726 {
727 outputFile << "TENSORS ";
728 ExposeMetaData<StringType>(metaDic, "pointTensorDataName", dataName);
729 outputFile << dataName << " ";
730 break;
731 }
732 case IOPixelEnum::ARRAY:
733 case IOPixelEnum::VARIABLELENGTHVECTOR:
734 {
735 outputFile << "COLOR_SCALARS ";
736 ExposeMetaData<StringType>(metaDic, "pointColorScalarDataName", dataName);
737 outputFile << dataName << " ";
738 WriteColorScalarBufferAsASCII(
739 outputFile, buffer, this->m_NumberOfPointPixelComponents, this->m_NumberOfPointPixels);
740 return;
741 }
742 default:
743 {
744 itkExceptionMacro(<< "Unknown point pixel type");
745 }
746 }
747
748 outputFile << pointPixelComponentName << '\n';
749
750 if (this->m_PointPixelType == IOPixelEnum::SCALAR)
751 {
752 outputFile << "LOOKUP_TABLE default" << '\n';
753 }
754
755 Indent indent(2);
756 if (this->m_PointPixelType == IOPixelEnum::SYMMETRICSECONDRANKTENSOR)
757 {
758 T * ptr = buffer;
759 SizeValueType i = 0;
760 const SizeValueType num = this->m_NumberOfPointPixelComponents * this->m_NumberOfPointPixels;
761 // Note that only the 3D tensors are supported in the VTK File Format
762 // documentation.
763 if (this->m_NumberOfPointPixelComponents == 3)
764 {
765 T zero(itk::NumericTraits<T>::ZeroValue());
766 T e12;
767 while (i < num)
768 {
769 // row 1
770 outputFile << ConvertNumberToString(*ptr++) << indent;
771 e12 = *ptr++;
772 outputFile << ConvertNumberToString(e12) << indent;
773 outputFile << ConvertNumberToString(zero) << '\n';
774 // row 2
775 outputFile << ConvertNumberToString(e12) << indent;
776 outputFile << ConvertNumberToString(*ptr++) << indent;
777 outputFile << ConvertNumberToString(zero) << '\n';
778 // row 3
779 outputFile << ConvertNumberToString(zero) << indent << ConvertNumberToString(zero) << indent
780 << ConvertNumberToString(zero) << "\n\n";
781 i += 3;
782 }
783 }
784 else if (this->m_NumberOfPointPixelComponents == 6)
785 {
786 T e12;
787 T e13;
788 T e23;
789 while (i < num)
790 {
791 // row 1
792 outputFile << ConvertNumberToString(*ptr++) << indent;
793 e12 = *ptr++;
794 outputFile << ConvertNumberToString(e12) << indent;
795 e13 = *ptr++;
796 outputFile << ConvertNumberToString(e13) << '\n';
