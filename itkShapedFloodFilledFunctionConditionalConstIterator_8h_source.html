200  for (unsigned int i = 0; i < m_Seeds.size(); ++i)

202  if (this->m_Image->GetBufferedRegion().IsInside(m_Seeds[i]) && this->IsPixelIncluded(m_Seeds[i]))
203  {
204  // Push the seed onto the queue
