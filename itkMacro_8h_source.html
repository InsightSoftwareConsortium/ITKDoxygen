402 //
403 // A macro to disallow the copy constructor, copy assignment,
404 // move constructor, and move assignment functions.
405 // This should be used in the public: declarations for a class
406 //
407 // ITK's paradigm for smart pointer and pipeline consistency
408 // prohibits the use of copy/move construction and copy/move assignment
409 // functions.

411 #define ITK_DISALLOW_COPY_AND_MOVE(TypeName) \
412  TypeName(const TypeName &) = delete; \
413  TypeName & operator=(const TypeName &) = delete; \
414  TypeName(TypeName &&) = delete; \
415  TypeName & operator=(TypeName &&) = delete

608 // Setup legacy code policy.
609 //
610 // CMake options:
611 // - When ITK_LEGACY_REMOVE:BOOL=ON, legacy code is hidden, thus causing compiler errors for code that depends on it
612 // - When ITK_LEGACY_REMOVE:BOOL=OFF, and ITK_LEGACY_SILENT:BOOL=ON, use
613 // of legacy code will not produce compiler warnings.
614 // - When ITK_LEGACY_REMOVE:BOOL=OFF, and ITK_LEGACY_SILENT:BOOL=OFF, use
615 // of legacy code will produce compiler warnings
616 //
617 // ITK_LEGACY_SILENT silently use legacy code. The default is to warn about legacy code use.
618 //
619 // Source files that test the legacy code may define ITK_LEGACY_TEST
620 // like this:
621 //
622 // #define ITK_LEGACY_TEST
623 // #include "itkClassWithDeprecatedMethod.h"
624 //
625 // in order to silence the warnings for calling deprecated methods.
626 // No other source files in ITK should call the methods since they are
627 // provided only for compatibility with older user code.
628 
629 // Define itkLegacyMacro to mark legacy methods where they are
630 // declared in their class. Example usage:
631 //
632 // // \deprecated Replaced by MyOtherMethod() as of ITK 2.0.
633 // itkLegacyMacro(void MyMethod());
634 //
635 // See below for what to do for the method definition.
636 #if defined(ITK_LEGACY_REMOVE)
637 # define itkLegacyMacro(method) /* no ';' */
638 #else
639 # if defined(ITK_LEGACY_SILENT) || defined(ITK_LEGACY_TEST)
640 // Provide legacy methods with no warnings.
641 # define itkLegacyMacro(method) method
642 # else
643 // Request compile-time warnings for uses of deprecated methods.
644 # define itkLegacyMacro(method) [[deprecated]] method
645 # endif
646 #endif
