58class AbsoluteValueDifference2
59{
60public:

72 inline TOutput
73 operator()(const TInput1 & A, const TInput2 & B) const
74 {
75 const auto dA = static_cast<double>(A);
76 const auto dB = static_cast<double>(B);
77 const double diff = dA - dB;
