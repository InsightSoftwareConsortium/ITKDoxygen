218 OffsetValueType
219 FindRoot(OffsetValueType x)
220 {
221 if (m_Parent[x] >= 0)
222 {
223 m_Parent[x] = FindRoot(m_Parent[x]);
224 return (m_Parent[x]);
225 }
226 else
227 {
228 return (x);
229 }
230 }
231
232 bool
233 Criterion(OffsetValueType x, OffsetValueType y)
234 {
235 return ((m_Raw[x] == m_Raw[y]) || (m_AuxData[x] < m_Lambda));
236 }
237
238 void
239 Union(OffsetValueType n, OffsetValueType p)
240 {
241 OffsetValueType r = FindRoot(n);
242
243 if (r != p)
244 {
245 if (Criterion(r, p))
246 {
247 m_AuxData[p] += m_AuxData[r];
248 m_Parent[r] = p;
249 }
250 else
251 {
252 m_AuxData[p] = m_Lambda;
253 }
254 }
255 }
