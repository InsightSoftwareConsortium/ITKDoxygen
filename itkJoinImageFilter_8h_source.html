47template <typename TPixel1, typename TPixel2>

49{
50public:

52 ~JoinFunctor() = default;
56 using Self = JoinFunctor;
57
59 using ValueType1 = typename PixelTraits<TPixel1>::ValueType;
60 using ValueType2 = typename PixelTraits<TPixel2>::ValueType;
61 using JoinValueType = typename JoinTraits<ValueType1, ValueType2>::ValueType;
62
64 static constexpr unsigned int Dimension1 = PixelTraits<TPixel1>::Dimension;
65 static constexpr unsigned int Dimension2 = PixelTraits<TPixel2>::Dimension;
66 static constexpr unsigned int JoinDimension = Dimension1 + Dimension2;
67
69 using JoinType = Vector<JoinValueType, Self::JoinDimension>;
70
71 bool
72 operator==(const JoinFunctor &) const
73 {
74 return true;
75 }
76
77 ITK_UNEQUAL_OPERATOR_MEMBER_FUNCTION(JoinFunctor);
78
80 inline JoinType
81 operator()(const TPixel1 & A, const TPixel2 & B) const
82 {
83 JoinType out;
84
85 // Copy A into the output, casting as necessary
86 this->FirstCopier(out, 0, A);
87
88 // Copy B into the output, starting where A left off,casting as necessary
89 this->SecondCopier(out, Dimension1, B);
90
91 return out;
92 }

101 struct CopierDispatchBase
102 {};
103 template <unsigned int VDimension>
104 struct CopierDispatch : public CopierDispatchBase
105 {};

117 void
118 FirstCopier(JoinType & out, unsigned int idx, const TPixel1 & A) const

121 }
