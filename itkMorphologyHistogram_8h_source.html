30template <typename TInputPixel, typename TCompare>
31class MorphologyHistogram
32{
33public:
34 using MapType = typename std::map<TInputPixel, IdentifierType, TCompare>;
35
36 MorphologyHistogram() = default;
37
38 inline void
39 AddBoundary()
40 {
41 m_Map[m_Boundary]++;
42 }
43
44 inline void
45 RemoveBoundary()
46 {
47 m_Map[m_Boundary]--;
48 }
49
50 inline void
51 AddPixel(const TInputPixel & p)
52 {
53 m_Map[p]++;
54 }
55
56 inline void
57 RemovePixel(const TInputPixel & p)
58 {
59 m_Map[p]--;
60 }
61
62 inline TInputPixel
63 GetValue()
64 {
65 itkAssertInDebugAndIgnoreInReleaseMacro(!m_Map.empty());
66 // clean the map
67 auto mapIt = m_Map.begin();
68 while (mapIt != m_Map.end())
69 {
70 if (mapIt->second == 0)
71 {
72 // this value must be removed from the histogram
73 // The value must be stored and the iterator updated before removing the
74 // value
75 // or the iterator is invalidated.
76 TInputPixel toErase = mapIt->first;
77 ++mapIt;
78 m_Map.erase(toErase);
79 }
80 else
81 {
82 ++mapIt;
83 // don't remove all the zero value found, just remove the one before the
84 // current maximum value
85 // the histogram may become quite big on real type image, but it's an
86 // important increase of performances
87 break;
88 }
89 }
90
91 // and return the value
92 itkAssertInDebugAndIgnoreInReleaseMacro(!m_Map.empty());
93 return m_Map.begin()->first;
94 }
95
96 inline TInputPixel
97 GetValue(const TInputPixel &)
98 {
99 return GetValue();
100 }
101
102 void
103 SetBoundary(const TInputPixel & val)
104 {
105 m_Boundary = val;
106 }
107
108 static bool
109 UseVectorBasedAlgorithm()
110 {
111 return false;
112 }
113
114 MapType m_Map;
115 TInputPixel m_Boundary;
116};

118template <typename TInputPixel, typename TCompare>
119class VectorMorphologyHistogram
120{
121public:
122 VectorMorphologyHistogram()
123 {
124 // initialize members need for the vector based algorithm
125 m_Vector.resize(NumericTraits<TInputPixel>::max() - NumericTraits<TInputPixel>::NonpositiveMin() + 1, 0);
126 if (m_Compare(NumericTraits<TInputPixel>::max(), NumericTraits<TInputPixel>::NonpositiveMin()))
127 {
128 m_InitValue = NumericTraits<TInputPixel>::NonpositiveMin();
129 m_CurrentValue = m_InitValue;
130 m_Direction = -1;
131 }
132 else
133 {
134 m_InitValue = NumericTraits<TInputPixel>::max();
135 m_CurrentValue = m_InitValue;
136 m_Direction = 1;
137 }
138 m_Boundary = 0;
139 }
140
141 inline void
142 AddBoundary()
143 {
144 AddPixel(m_Boundary);
145 }
146
147 inline void
148 RemoveBoundary()
149 {
150 RemovePixel(m_Boundary);
151 }
152
153 inline void
154 AddPixel(const TInputPixel & p)
155 {
156 m_Vector[p - NumericTraits<TInputPixel>::NonpositiveMin()]++;
157 if (m_Compare(p, m_CurrentValue))
158 {
159 m_CurrentValue = p;
160 }
161 }
162
163 inline void
164 RemovePixel(const TInputPixel & p)
165 {
166 m_Vector[p - NumericTraits<TInputPixel>::NonpositiveMin()]--;
167 while (m_Vector[m_CurrentValue - NumericTraits<TInputPixel>::NonpositiveMin()] == 0 &&
168 m_CurrentValue != m_InitValue)
169 {
170 m_CurrentValue += m_Direction;
171 }
172 }
173
174 inline TInputPixel
175 GetValue()
176 {
177 return m_CurrentValue;
178 }
179
180 inline TInputPixel
181 GetValue(const TInputPixel &)
182 {
183 return GetValue();
184 }
185
186 void
187 SetBoundary(const TInputPixel & val)
188 {
189 m_Boundary = val;
190 }
191
192 static bool
193 UseVectorBasedAlgorithm()
194 {
195 return true;
196 }
197
198 std::vector<IdentifierType> m_Vector;
199 TInputPixel m_InitValue;
200 TInputPixel m_CurrentValue;
201 TCompare m_Compare;
202 int m_Direction;
203 TInputPixel m_Boundary;
204};
