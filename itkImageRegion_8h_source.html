286 template <typename TCoordRepType>
287 bool
288 IsInside(const ContinuousIndex<TCoordRepType, VImageDimension> & index) const
289 {
290 constexpr TCoordRepType half = 0.5;
291 for (unsigned int i = 0; i < ImageDimension; ++i)
292 {
293 // Use negation of tests so that index[i]==NaN leads to returning false.
294 if (!(index[i] >= m_Index[i] - half && index[i] <= (m_Index[i] + static_cast<IndexValueType>(m_Size[i])) - half))
295 {
296 return false;
297 }
298 }
299 return true;

307 bool
308 IsInside(const Self & otherRegion) const
309 {
310 const auto & otherIndex = otherRegion.m_Index;
311 const auto & otherSize = otherRegion.m_Size;
314 for (unsigned int i = 0; i < ImageDimension; ++i)
315 {
316 if (otherIndex[i] < m_Index[i] || otherSize[i] == 0 ||
317 otherIndex[i] + static_cast<IndexValueType>(otherSize[i]) >
318 m_Index[i] + static_cast<IndexValueType>(m_Size[i]))
319 {
320 return false;
321 }
322 }
323 return true;
