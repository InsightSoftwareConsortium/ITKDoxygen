242  bool
243  IsInside(const IndexType & index) const
244  {
245  for (unsigned int i = 0; i < ImageDimension; ++i)
246  {
247  if (index[i] < m_Index[i])
248  {
249  return false;
250  }
251  if (index[i] >= (m_Index[i] + static_cast<IndexValueType>(m_Size[i])))
252  {

254  }
255  }

264  template <typename TCoordRepType>
265  bool
266  IsInside(const ContinuousIndex<TCoordRepType, VImageDimension> & index) const
267  {
268  for (unsigned int i = 0; i < ImageDimension; ++i)
269  {
270  if (Math::RoundHalfIntegerUp<IndexValueType>(index[i]) < static_cast<IndexValueType>(m_Index[i]))
271  {
272  return false;
273  }
274  // bound is the last valid pixel location
275  const auto bound = static_cast<TCoordRepType>(m_Index[i] + m_Size[i] - 0.5);
277 
278  /* Note for NaN: test using negation of a positive test in order
279  * to always evaluate to true (and thus return false) when index[i]
280  * is NaN. The cast above to integer via RoundHalfIntegerUp will cast
281  * NaN into a platform-dependent value (large negative, -1 or large
282  * positive, empirically). Thus this test here is relied on
283  * to 'catch' NaN's. */
284  if (!(index[i] <= bound))
285  {
286  return false;
287  }
288  }
289  return true;
290  }
